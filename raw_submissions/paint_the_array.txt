 which yields a beautiful coloring, or report that it is impossible.

The second line of each testcase contains n

 (1≤ai≤1018
).

Output
For each testcase print a single integer. If there is no such value of d
 that yields a beautiful coloring, print 0
. Otherwise, print any suitable value of d
 (1≤d≤1018
).

void solve() {
    cin >> n;
    vector<int> v(n);
    for (auto &i: v) cin >> i;
    vector<int> check = {1, 2};

    for (int i = 0; i < n; i++)
        for (int j = i; j < n; j++) {
            if (abs(i - j) != 1)
                check.pb({__gcd(v[i], v[j])});
        }
    sort(all(check));
    check.resize(unique(all(check)) - check.begin());
    for (auto i: check) {
        vector<int> col(n, 1);
        bool ok = true;
        for (int j = 1; j < n; j++) {
            int ost = v[j] % i, ost2 = v[j - 1] % i;
            if ((ost == 0 && ost2 == 0) || (ost != 0 && ost2 != 0)) {
                ok = false;
                break;
            }
        }
        if (ok) {
            cout << i << '\n';
            return;
        }
    }
    cout << 0 << '\n';
}
